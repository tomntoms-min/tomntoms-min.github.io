rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    match /rankings/{rankingId} {
      allow read: if true;
      allow create: if
        request.resource.data.keys().hasOnly(['name', 'score', 'served', 'createdAt', 'clientDate']) &&
        request.resource.data.name is string &&
        request.resource.data.name.size() >= 1 &&
        request.resource.data.name.size() <= 12 &&
        request.resource.data.score is int &&
        request.resource.data.score >= 0 &&
        request.resource.data.score <= 100000 &&
        request.resource.data.served is int &&
        request.resource.data.served >= 0 &&
        request.resource.data.served <= 100 &&
        request.resource.data.createdAt == request.time &&
        request.resource.data.clientDate is string;
      allow update, delete: if false;
    }
  }
}
